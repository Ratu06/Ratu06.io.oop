Program 7

Encapsulation

Metode perlindungan data dilakukan dengan membatasi akses langsung ke variabel internal melalui penggunaan awalan underscore (_).

// Nama   : Ratu klarisa
// Nim    : 251410010
// Kelas  : Si2A

class Person {
  String _name;
  int age;

  Person(this._name,this.age);

  void setName(String name) => _name = name;

  String getName() => _name;
}

void main(){
  Person person =
  Person("Ratu Klarisa",19);

  print(person.age);
  person.setName("Ratu Klarisa");

  print(person.getName());
}

Program 9

Product Class

Penggunaan Named Parameters dan Required untuk menciptakan objek yang lebih deskriptif dan aman.

// Nama   : Ratu klarisa
// Nim    : 251410010
// Kelas  : Si2A

class Produk {
  final String id, nama;
  final double harga;
  final int stok;
  final string kategori;

  Produk({
    required this.id,required this.nama,required this.harga,this.stok = 0
    this.kategori = 'Umum',
  })
}

void main() {
  Produk laptop = Produk(
    id: "P001"
    nama: "Laptop Gaming",
    harga: 15000000,
    stok: 15,
    kategori: "Elektronik",
  );

  Produk smartphone = Produk(
    id: "P002",
    nama: "Smartphone Flagship",
    harga: 12000000,
    stok: 25,
  );
  
  Produk kamera = Produk(
    id: "P003",
    nama: "Kamera Mirrorles",
    harga: 20000000,
  );

  print(laptop.nama);
  print(smartphone.nama);
  print(kamera.nama);
}
